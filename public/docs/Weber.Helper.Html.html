Weber.Helper.Html

HTML generator helper.

## Example

# Generates <p class="class_test">test</p>
def do_something do
  tag(:p, "test", [class: "class_test"])
end

# Generates <div id="test"><p>test</p></div>
def do_something do
  tag(:div, [id: "test"]) do
    tag(:p, "test")
  end
end
